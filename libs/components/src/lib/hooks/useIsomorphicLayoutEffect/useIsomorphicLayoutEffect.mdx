### How it works

A react hook to use `useLayoutEffect` on the client and `useEffect` on the server.

````jsx
import { useIsomorphicLayoutEffect } from '@transavia/take-off-ui'

export default function Component() {
  useIsomorphicLayoutEffect(() => {
    console.log(
      "In the browser, I'm an `useLayoutEffect`, but in SSR, I'm an `useEffect`.",
    )
  }, [])

  return <p>Hello, world</p>
}
````
